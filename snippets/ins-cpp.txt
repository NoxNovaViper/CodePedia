#include <iostream>
using namespace std;

int main() {
    // 1. Define the Array (size 10 to allow growth)
    int arr[10] = {10, 20, 30, 40, 50};
    int n = 5; // current elements
    int pos = 2, val = 99;

    // 2. Shift elements right
    for (int i = n; i > pos; i--) {
        arr[i] = arr[i - 1];
    }

    // 3. Insert
    arr[pos] = val;
    n++; // update count

    for(int i=0; i<n; i++) cout << arr[i] << " ";
    return 0;
}
