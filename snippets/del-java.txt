public class Main {
    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50};
        int n = 5, pos = 2;

        // Shift left
        for (int i = pos; i < n - 1; i++) {
            arr[i] = arr[i + 1];
        }

        // In Java, fixed arrays don't shrink, 
        // so we just print up to n-1
        for (int i = 0; i < n - 1; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}
