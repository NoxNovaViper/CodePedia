let arr = [10, 20, 30, 40, 50];
let target = 40;
let low = 0, high = arr.length - 1;
let foundAt = -1;
while (low <= high) {
    let mid = Math.floor((low + high) / 2);
    if (arr[mid] === target) {
        foundAt = mid;
        break;
    } else if (arr[mid] < target) {
        low = mid + 1;
    } else {
        high = mid - 1;
    }
}
console.log("Index:", foundAt);
