public class Main {
    public static void main(String[] args) {
        // 1. Define the Array
        int[] arr = {15, 72, 8, 91, 23, 56};

        // 2. Find Max
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
        }

        System.out.println("Maximum Value: " + max);
    }
}
