arr := []int{10, 20, 30, 40, 50, 60}
target := 40
low, high := 0, len(arr)-1
foundAt := -1

for low <= high {
    mid := low + (high-low)/2
    if arr[mid] == target {
        foundAt = mid
        break
    } else if arr[mid] < target {
        low = mid + 1
    } else {
        high = mid - 1
    }
}
