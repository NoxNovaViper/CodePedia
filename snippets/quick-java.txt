int[] arr = {10, 80, 30, 90, 40, 50, 70};
int pivot = arr[arr.length - 1];
int i = -1;

for (int j = 0; j < arr.length - 1; j++) {
    if (arr[j] < pivot) {
        i++;
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
int temp = arr[i + 1];
arr[i + 1] = arr[arr.length - 1];
arr[arr.length - 1] = temp;
