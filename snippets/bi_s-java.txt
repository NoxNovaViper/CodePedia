int[] arr = {10, 20, 30, 40, 50, 60};
int target = 40;
int low = 0, high = arr.length - 1;
int foundAt = -1;

while (low <= high) {
    int mid = low + (high - low) / 2;
    if (arr[mid] == target) {
        foundAt = mid;
        break;
    } else if (arr[mid] < target) {
        low = mid + 1;
    } else {
        high = mid - 1;
    }
}
