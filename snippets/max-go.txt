package main
import "fmt"

func main() {
    // 1. Define the Array
    arr := []int{15, 72, 8, 91, 23, 56}

    // 2. Find Max
    max := arr[0]
    for i := 1; i < len(arr); i++ {
        if arr[i] > max {
            max = arr[i]
        }
    }

    fmt.Println("Maximum Value:", max)
}
