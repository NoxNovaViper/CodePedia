let arr = [10, 80, 30, 90, 40, 50, 70];
let pivot = arr[arr.length - 1]; 
let i = -1; 

for (let j = 0; j < arr.length - 1; j++) {
    if (arr[j] < pivot) {
        i++;
        [arr[i], arr[j]] = [arr[j], arr[i]]; 
    }
}
[arr[i + 1], arr[arr.length - 1]] = [arr[arr.length - 1], arr[i + 1]];
console.log(arr);
