#include <iostream>
using namespace std;

int main() {
    // 1. Define the Array
    int arr[] = {10, 20, 30, 40, 50};
    int n = 5;
    int pos = 2; // Index to delete

    // 2. Shift left
    for (int i = pos; i < n - 1; i++) {
        arr[i] = arr[i + 1];
    }

    // 3. Decrement size
    n--;

    for(int i=0; i<n; i++) cout << arr[i] << " ";
    return 0;
}
