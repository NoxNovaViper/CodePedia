func lcm(a, b int) int {
    if a == 0 || b == 0 { return 0 }
    return (a * b) / gcd(a, b)
}


func lcm(a, b int) int64 {
    if a == 0 || b == 0 { return 0 }
    x, y := abs(a), abs(b)
    for y != 0 {
        x, y = y, x%y
    }
    return int64(abs(a) / x) * int64(abs(b))
}

func abs(x int) int {
    if x < 0 { return -x }
    return x
}
