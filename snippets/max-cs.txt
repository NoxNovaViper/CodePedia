using System;

class Program {
    static void Main() {
        // 1. Define the Array
        int[] arr = { 15, 72, 8, 91, 23, 56 };

        // 2. Find Max
        int max = arr[0];
        for (int i = 1; i < arr.Length; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
        }

        Console.WriteLine("Maximum Value: " + max);
    }
}
