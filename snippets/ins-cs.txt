using System;

class Program {
    static void Main() {
        int[] arr = new int[10] { 10, 20, 30, 40, 50, 0, 0, 0, 0, 0 };
        int n = 5, pos = 2, val = 99;

        // Shift right
        for (int i = n; i > pos; i--) {
            arr[i] = arr[i - 1];
        }

        arr[pos] = val;
        n++;

        Console.WriteLine(string.Join(", ", arr[..n]));
    }
}
