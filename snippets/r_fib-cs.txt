using System;

class Program {
    // Recursive Function
    static int Fibonacci(int n) {
        if (n <= 1) return n;
        return Fibonacci(n - 1) + Fibonacci(n - 2);
    }

    static void Main() {
        int n = 6;
        Console.WriteLine($"Fibonacci at position {n} is {Fibonacci(n)}");
    }
}
