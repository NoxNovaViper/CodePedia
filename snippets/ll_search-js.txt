class Node {
    constructor(data) {
        this.data = data;
        this.next = null;
    }
}

// 1. Setup
let head = new Node(10); head.next = new Node(20);
let target = 20;

// 2. Search Logic
let current = head;
let found = false;
while (current !== null) {
    if (current.data === target) {
        found = true;
        break;
    }
    current = current.next;
}
console.log("Found:", found);
