function deleteNode(head, key) {
    let temp = head, prev = null;

    if (temp !== null && temp.data === key) {
        return temp.next; // Changed head
    }

    while (temp !== null && temp.data !== key) {
        prev = temp;
        temp = temp.next;
    }

    if (temp === null) return head;

    prev.next = temp.next; // Bypass the node
    return head;
}
