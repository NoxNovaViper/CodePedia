class Node {
    int data;
    Node next;
    Node(int d) { data = d; next = null; }
}

public class Main {
    public static void main(String[] args) {
        // 1. Setup
        Node head = new Node(10);
        head.next = new Node(20);
        head.next.next = new Node(30);

        // 2. Traverse
        Node current = head;
        while (current != null) {
            System.out.println(current.data);
            current = current.next;
        }
    }
}
