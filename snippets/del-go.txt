package main
import "fmt"

func main() {
    arr := []int{10, 20, 30, 40, 50}
    pos := 2

    // Shift left
    for i := pos; i < len(arr)-1; i++ {
        arr[i] = arr[i+1]
    }

    // Slice off the last element
    arr = arr[:len(arr)-1]

    fmt.Println("After Deletion:", arr)
}
