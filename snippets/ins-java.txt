public class Main {
    public static void main(String[] args) {
        // 1. Define the Array
        int[] arr = new int[10];
        arr[0]=10; arr[1]=20; arr[2]=30; arr[3]=40; arr[4]=50;
        int n = 5, pos = 2, val = 99;

        // 2. Shift right
        for (int i = n; i > pos; i--) {
            arr[i] = arr[i - 1];
        }

        // 3. Insert
        arr[pos] = val;
        n++;

        for(int i=0; i<n; i++) System.out.print(arr[i] + " ");
    }
}
