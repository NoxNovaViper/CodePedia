using System;

class Program {
    static void Main() {
        // 1. Define the Array
        int[] arr = { 10, 20, 30, 40, 50 };

        // 2. Reverse
        int left = 0;
        int right = arr.Length - 1;
        while (left < right) {
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            
            left++;
            right--;
        }

        Console.WriteLine(string.Join(", ", arr));
    }
}
