class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

# 1. Create Nodes
head = Node(10)
head.next = Node(20)
head.next.next = Node(30)

# 2. Manual Traverse
current = head
while current is not None:
    print(current.data)
    current = current.next # Move to next node
