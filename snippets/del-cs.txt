using System;

class Program {
    static void Main() {
        int[] arr = { 10, 20, 30, 40, 50 };
        int n = 5, pos = 2;

        // Shift left
        for (int i = pos; i < n - 1; i++) {
            arr[i] = arr[i + 1];
        }

        // Print new logical size
        for (int i = 0; i < n - 1; i++) {
            Console.Write(arr[i] + " ");
        }
    }
}
