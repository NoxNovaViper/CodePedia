package main
import "fmt"
func main() {
    var n int
    fmt.Scan(&n)
    isPrime := n > 1
    for i := 2; i * i <= n; i++ {
        if n % i == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println("Prime")
    } else {
        fmt.Println("Not Prime")
    }
}
